# Router Interfaces

interface Fa0/0
ip address 1.0.0.1 255.0.0.0
no shutdown

interface Eth0/0/0
ip address 2.0.0.1 255.0.0.0
no shutdown

interface Fa0/1
ip address 3.0.0.1 255.0.0.0
no shutdown

interface Eth0/1/0
ip address 4.0.0.1 255.0.0.0
no shutdown

# DHCP

ip dhcp pool LAN1
network 1.0.0.0 255.0.0.0
default-router 1.0.0.1
